# calculators/ownership.py
from sqlalchemy.orm import Session
from models import OwnershipData, OwnershipCalculations, Scenario
from typing import Dict, Any
import logging

logger = logging.getLogger(__name__)

class OwnershipCalculator:
    """Complete B-BBEE Ownership calculator matching Excel formulas exactly"""
    
    def __init__(self, db: Session):
        self.db = db
        self.graduation_table = {
            1: 0.1, 2: 0.2, 3: 0.4, 4: 0.6,
            5: 0.8, 6: 1.0, 7: 1.0, 8: 1.0,
            9: 1.0, 10: 1.0
        }
        self.target_economic_interest = 0.25  # 25% target
        self.max_points = {
            'voting_rights': 4,
            'economic_interest': 8,
            'net_value': 8,
            'total': 25
        }
    
    def calculate_graduation_factor(self, years_held: float) -> float:
        """Calculate graduation factor based on years held"""
        if years_held <= 0:
            return 0
        
        # Find appropriate factor from table (Excel VLOOKUP with TRUE)
        years = sorted(self.graduation_table.keys())
        factor = 0
        for year in years:
            if year <= years_held:
                factor = self.graduation_table[year]
            else:
                break
        return factor
    
    def calculate_formula_a(self, economic_interest: float, years_held: float) -> float:
        """
        Formula A: Time-based graduation
        = B × (1 / (Target_Economic_Interest × Graduation_Factor)) × 8
        """
        grad_factor = self.calculate_graduation_factor(years_held)
        if grad_factor == 0:
            return 0
        
        denominator = self.target_economic_interest * grad_factor
        if denominator == 0:
            return 0
        
        result = economic_interest * (1 / denominator) * 8
        return min(result, self.max_points['economic_interest'])
    
    def calculate_formula_b(self, economic_interest: float) -> float:
        """
        Formula B: Straight economic interest
        = Economic_Interest_% / Target_Economic_Interest × 8
        """
        if self.target_economic_interest == 0:
            return 0
        
        result = (economic_interest / self.target_economic_interest) * 8
        return min(result, self.max_points['economic_interest'])
    
    def calculate_deemed_value(self, company_value: float, outstanding_debt: float, 
                              carrying_value: float) -> float:
        """
        Deemed Value after debt
        = (Value_of_Equity - Outstanding_Debt) / Carrying_Value_of_Debt
        """
        if carrying_value == 0:
            return 0
        
        result = (company_value - outstanding_debt) / carrying_value
        return max(0, min(result, self.max_points['net_value']))
    
    def calculate_voting_rights(self, shareholders_data: List[Dict]) -> Dict[str, float]:
        """
        Calculate voting rights points including black women bonus
        """
        total_black_ownership = sum(s.get('black_ownership_percentage', 0) for s in shareholders_data)
        total_black_women_ownership = sum(
            s.get('black_women_ownership_percentage', 0) 
            for s in shareholders_data if s.get('gender') == 'female'
        )
        
        # Voting rights (max 4)
        voting_rights = min(total_black_ownership * 4, 4)
        
        # Black women bonus (max 2)
        women_bonus = min(total_black_women_ownership * 2, 2)
        
        return {
            'voting_rights': voting_rights,
            'women_bonus': women_bonus,
            'total_voting': voting_rights + women_bonus
        }
    
    def calculate_economic_interest(self, shareholders_data: List[Dict]) -> Dict[str, float]:
        """
        Calculate economic interest points using both formulas and taking best
        """
        total_economic_interest = sum(s.get('economic_interest_percentage', 0) for s in shareholders_data)
        avg_years_held = sum(s.get('years_held', 0) for s in shareholders_data) / max(len(shareholders_data), 1)
        
        formula_a_points = self.calculate_formula_a(total_economic_interest, avg_years_held)
        formula_b_points = self.calculate_formula_b(total_economic_interest)
        
        # Take the better of Formula A and B
        economic_points = max(formula_a_points, formula_b_points)
        
        return {
            'formula_a': formula_a_points,
            'formula_b': formula_b_points,
            'economic_points': economic_points
        }
    
    def calculate_net_value(self, shareholders_data: List[Dict], company_value: float,
                           outstanding_debt: float) -> float:
        """
        Calculate net value points
        """
        total_shares = sum(s.get('shares', 0) for s in shareholders_data)
        if total_shares == 0:
            return 0
        
        # Calculate weighted net value
        weighted_net_value = 0
        for shareholder in shareholders_data:
            share_percentage = shareholder.get('shares', 0) / total_shares
            debt_attributable = outstanding_debt * share_percentage
            carrying_value = shareholder.get('share_value', 0) * share_percentage
            
            deemed_value = self.calculate_deemed_value(
                company_value * share_percentage,
                debt_attributable,
                carrying_value
            )
            weighted_net_value += deemed_value
        
        return min(weighted_net_value, self.max_points['net_value'])
    
    def check_sub_minimum(self, voting_points: float, economic_points: float,
                         net_value_points: float) -> bool:
        """
        Check sub-minimum requirements (40% of net value points)
        Net value sub-minimum = 40% of 8 = 3.2 points
        """
        return net_value_points >= 3.2
    
    def calculate_full_scorecard(self, scenario_id: str) -> Dict[str, Any]:
        """
        Calculate complete ownership scorecard for a scenario
        """
        # Get all ownership data for this scenario
        shareholders = self.db.query(OwnershipData).filter(
            OwnershipData.scenario_id == scenario_id,
            OwnershipData.is_active == True
        ).all()
        
        # Get company data
        calc_data = self.db.query(OwnershipCalculations).filter(
            OwnershipCalculations.scenario_id == scenario_id
        ).first()
        
        if not shareholders or not calc_data:
            raise ValueError(f"No ownership data found for scenario {scenario_id}")
        
        # Prepare shareholder data for calculations
        shareholders_data = []
        for s in shareholders:
            shareholders_data.append({
                'black_ownership_percentage': s.share_percentage if s.is_black_designated_group else 0,

            # Get participants for this program
            for participant in program.get('participants', []):
                if participant.get('is_black', False):
                    # Calculate cost per participant (proportional)
                    participant_cost = program['cost'] / max(len(program.get('participants', [])), 1)
                    
                    overall_black += participant_cost
                    
                    if program['category'] == 'bursary':
                        bursaries += participant_cost
                    
                    if participant.get('is_employed', False) == False:
                        absorption += participant_cost
                    
                    if participant.get('is_disabled', False):
                        disability += participant_cost
        
        return {
            'overall_black': overall_black,
            'bursaries': bursaries,
            'absorption': absorption,
            'disability': disability
        }

        if targets['overall_black'] > 0:
            skills_points = min(
                self.max_points['skills_spend'],
                (actual['overall_black'] / targets['overall_black']) * self.max_points['skills_spend']
            )
        else:
            skills_points = 0

                self.max_points['bursaries'],
                (actual['bursaries'] / targets['bursaries']) * self.max_points['bursaries']
            )
        else:
            bursary_points = 0
        
        # Absorption points (max 5)
        if targets['absorption'] > 0:
            absorption_points = min(
                self.max_points['absorption'],
                (actual['absorption'] / targets['absorption']) * self.max_points['absorption']
            )
        else:
            absorption_points = 0
        
        return {
            'skills_spend': skills_points,
            'bursaries': bursary_points,
            'absorption': absorption_points,
            'total': skills_points + bursary_points + absorption_points
        }

# calculators/procurement.py
class ProcurementCalculator:
    """Complete Preferential Procurement calculator"""
    
    def __init__(self, db: Session):
        self.db = db
        self.recognition_table = {
            1: 1.35,  # Level 1: 135%
            2: 1.25,  # Level 2: 125%
            3: 1.10,  # Level 3: 110%
            4: 1.00,  # Level 4: 100%
            5: 0.80,  # Level 5: 80%
            6: 0.60,  # Level 6: 60%
            7: 0.50,  # Level 7: 50%
            8: 0.10   # Level 8: 10%
        }
        self.max_points = {
            'procurement': 25,
            'bonus': 2,
            'total': 27
        }
        self.bonus_threshold = 0.51  # 51% black-owned for bonus
    
    def calculate_recognised_spend(self, supplier: Dict) -> float:
        """
        Calculate recognised spend for a single supplier
        = Supplier_Spend × Recognition_Level × (Bonus if 51% black-owned)
        """
        # Base recognition
        recognition = self.recognition_table.get(supplier.get('bee_level', 8), 0)
        recognised = supplier['spend'] * recognition
        
        # 51% black-owned bonus
        if supplier.get('black_ownership', 0) >= self.bonus_threshold:
            recognised *= 1.15
            supplier['qualifies_for_bonus'] = True
        else:
            supplier['qualifies_for_bonus'] = False
        
        return recognised
    
    def calculate_tmps(self, cost_of_sales: float, operating_expenses: float,
                      exclusions: List[Dict]) -> float:
        """
        Calculate Total Measured Procurement Spend
        = Cost of Sales + Operating Expenses - Exclusions
        """
        total_exclusions = sum(e.get('amount', 0) for e in exclusions)
        tmps = cost_of_sales + operating_expenses - total_exclusions
        return max(0, tmps)
    
    def calculate_scorecard(self, scenario_id: str) -> Dict[str, Any]:
        """Calculate complete procurement scorecard"""
        
        # Get all suppliers for this scenario
        suppliers = self.db.query(Suppliers).filter(
            Suppliers.scenario_id == scenario_id
        ).all()
        
        # Get transactions for these suppliers
        transactions = self.db.query(ProcurementTransactions).filter(
            ProcurementTransactions.scenario_id == scenario_id
        ).all()
        
        # Get TMPS calculation
        calc_data = self.db.query(ProcurementCalculations).filter(
            ProcurementCalculations.scenario_id == scenario_id
        ).first()
        
        if not calc_data or not suppliers:
            raise ValueError(f"No procurement data found for scenario {scenario_id}")
        
        # Group transactions by supplier
        supplier_spend = {}
        for trans in transactions:
            supplier_id = trans.supplier_id
            if supplier_id not in supplier_spend:
                supplier_spend[supplier_id] = 0
            supplier_spend[supplier_id] += trans.amount
        
        # Calculate recognised spend for each supplier
        suppliers_data = []
        total_recognised = 0
        bonus_qualifying_spend = 0
        
        for supplier in suppliers:
            spend = supplier_spend.get(supplier.id, 0)
            supplier_dict = {
                'name': supplier.supplier_name,
                'bee_level': supplier.bee_level,
                'black_ownership': supplier.black_ownership_percentage,
                'spend': spend,
                'is_emi': supplier.is_emi,
                'is_qse': supplier.is_qse
            }
            
            recognised = self.calculate_recognised_spend(supplier_dict)
            total_recognised += recognised
            
            if supplier_dict.get('qualifies_for_bonus', False):
                bonus_qualifying_spend += spend
            
            supplier_dict['recognised_spend'] = recognised
            suppliers_data.append(supplier_dict)
        
        # Calculate procurement points
        if calc_data.tmps > 0:
            procurement_points = min(
                self.max_points['procurement'],
                (total_recognised / calc_data.tmps) * self.max_points['procurement']
            )
            
            bonus_points = min(
                self.max_points['bonus'],
                (bonus_qualifying_spend / calc_data.tmps) * self.max_points['bonus']
            )
        else:
            procurement_points = 0
            bonus_points = 0
        
        total_points = procurement_points + bonus_points
        
        # Check sub-minimum (40% of procurement points = 40% of 25 = 10)
        sub_minimum_met = procurement_points >= 10
        
        # Update database
        calc_data.total_recognised_spend = total_recognised
        calc_data.procurement_points = procurement_points
        calc_data.bonus_points = bonus_points
        calc_data.total_points = total_points
        calc_data.sub_minimum_met = sub_minimum_met
        calc_data.last_calculated = datetime.now()
        
        self.db.commit()
        
        return {
            'scenario_id': scenario_id,
            'tmps': calc_data.tmps,
            'total_recognised': total_recognised,
            'procurement_points': procurement_points,
            'bonus_points': bonus_points,
            'total_points': total_points,
            'sub_minimum_met': sub_minimum_met,
            'suppliers': suppliers_data,
            'utilization': (total_recognised / calc_data.tmps * 100) if calc_data.tmps > 0 else 0,
            'gap': max(0, calc_data.tmps - total_recognised),
            'max_points': self.max_points
        }

# calculators/summary.py
class SummaryCalculator:
    """Complete summary calculator with sub-minimum and discounting"""
    
    def __init__(self, db: Session):

        # Get YES
        yes = self.db.query(YESInitiatives).filter(

        ).all()
        
        return {
            'ownership': ownership,
            'management': management,
            'skills': skills,
            'procurement': procurement,
            'esd': esd,
            'sed': sed,
            'yes': yes
        }
    
    def check_sub_minimum(self, pillar_scores: Dict) -> Dict[str, bool]:
        """
        Check sub-minimum requirements for priority elements
        Returns dict of which pillars pass and overall status
        """
        results = {}
        
        # Ownership sub-minimum (net value >= 40% of 8 = 3.2)
        if pillar_scores['ownership']:
            results['ownership'] = pillar_scores['ownership'].sub_minimum_met
        else:
            results['ownership'] = False
        
        # Skills sub-minimum (skills spend >= 40% of 20 = 8)
        if pillar_scores['skills']:
            results['skills'] = pillar_scores['skills'].sub_minimum_met
        else:
            results['skills'] = False
        
        # Procurement sub-minimum (procurement >= 40% of 25 = 10)
        if pillar_scores['procurement']:
            results['procurement'] = pillar_scores['procurement'].sub_minimum_met
        else:
            results['procurement'] = False
        
        # Overall - ALL priority elements must pass
        results['overall'] = all(results[p] for p in self.priority_elements)
        
        return results
    
    def calculate_yes_contribution(self, yes_initiatives: List) -> float:
        """Calculate YES contribution points (max 2-3 depending on tier)"""
        total_points = 0
        
        for initiative in yes_initiatives:
            if initiative.yes_tier == 1:
                total_points += 2
            elif initiative.yes_tier == 2:
                total_points += 2
            elif initiative.yes_tier == 3:
                total_points += 3
            
            # Bonus for absorption
            if initiative.is_absorbed:
                total_points += 1
        
        return min(total_points, 3)  # Max 3 points
    
    def calculate_total_points(self, pillar_scores: Dict) -> float:
        """Calculate total points from all pillars"""
        total = 0
        
        if pillar_scores['ownership']:
            total += pillar_scores['ownership'].total_points or 0
        
        if pillar_scores['management']:
            total += pillar_scores['management'].total_points or 0
        
        if pillar_scores['skills']:
            total += pillar_scores['skills'].total_points or 0
        
        if pillar_scores['procurement']:
            total += pillar_scores['procurement'].total_points or 0
        
        if pillar_scores['esd']:
            total += pillar_scores['esd'].total_points or 0
        
        if pillar_scores['sed']:
            total += pillar_scores['sed'].total_points or 0
        
        # Add YES points
        yes_points = self.calculate_yes_contribution(pillar_scores['yes'])
        total += yes_points
        
        return total
    
    def points_to_level(self, points: float) -> int:
        """Convert points to B-BBEE level"""
        for threshold, level in sorted(self.level_table.items(), reverse=True):
            if points >= threshold:
                return level
        return 8
    
    def calculate_final_scorecard(self, client_id: str, scenario_id: str) -> Dict[str, Any]:
        """
        Calculate complete summary with discounting
        """
        # Get all pillar scores
        pillar_scores = self.get_pillar_scores(scenario_id)
        
        # Calculate total points
        total_points = self.calculate_total_points(pillar_scores)
        
        # Check sub-minimum
        sub_minimum_results = self.check_sub_minimum(pillar_scores)
        
        # Apply discount if sub-minimum fails
        if sub_minimum_results['overall']:
            discounted_points = total_points
            discount_applied = False
            discount_reason = None
        else:
            # Discount by 1 level (approximately 10 points)
            discounted_points = total_points - 10
            discount_applied = True
            failed_pillars = [p for p in self.priority_elements if not sub_minimum_results[p]]
            discount_reason = f"Sub-minimum failure in: {', '.join(failed_pillars)}"
        
        # Calculate levels
        achieved_level = self.points_to_level(total_points)
        discounted_level = self.points_to_level(discounted_points)
        
        # Get scenario details
        scenario = self.db.query(Scenario).filter(Scenario.id == scenario_id).first()
        
        return {
            'client_id': client_id,
            'scenario_id': scenario_id,
            'scenario_name': scenario.name if scenario else 'Base',
            'total_points': round(total_points, 2),
            'discounted_points': round(discounted_points, 2),
            'achieved_level': achieved_level,
            'discounted_level': discounted_level,
            'sub_minimum': sub_minimum_results,
            'discount_applied': discount_applied,
            'discount_reason': discount_reason,
            'pillar_breakdown': {
                'ownership': round(pillar_scores['ownership'].total_points, 2) if pillar_scores['ownership'] else 0,
                'management': round(pillar_scores['management'].total_points, 2) if pillar_scores['management'] else 0,
                'skills': round(pillar_scores['skills'].total_points, 2) if pillar_scores['skills'] else 0,
                'procurement': round(pillar_scores['procurement'].total_points, 2) if pillar_scores['procurement'] else 0,
                'esd': round(pillar_scores['esd'].total_points, 2) if pillar_scores['esd'] else 0,
                'sed': round(pillar_scores['sed'].total_points, 2) if pillar_scores['sed'] else 0,
                'yes': self.calculate_yes_contribution(pillar_scores['yes'])
            },
            'status': 'Compliant' if achieved_level <= 4 else 'Non-Compliant',
            'verification_status': 'Pass' if sub_minimum_results['overall'] and achieved_level <= 4 else 'Fail',
            'calculated_at': datetime.now().isoformat()
        }

# services/ai_ingestion.py
import openai
from langchain.document_loaders import PyPDFLoader, TextLoader, UnstructuredExcelLoader

class AIDocumentIngestion:
    """
    AI-powered document ingestion service
    Handles: Excel, PDF, Word, PPT, JPG
    """
    
    def __init__(self):
        self.llm = OpenAI(temperature=0)
        self.embeddings = OpenAIEmbeddings()
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=2000,
            chunk_overlap=200
        )
        
        # Schema for extraction (matches database structure)
        self.extraction_schema = {
            "properties": {
                "company_name": {"type": "string"},
                "registration_number": {"type": "string"},
                "financial_year_end": {"type": "string"},
                "revenue": {"type": "number"},

                    "type": "array",
                    "items": {
                        "type": "object",

                            "share_percentage": {"type": "number"},

                    }
                },

                    "type": "array",
                    "items": {
                        "type": "object",
                        "properties": {
                            "name": {"type": "string"},
                            "id_number": {"type": "string"},
                            "race": {"type": "string"},
                            "gender": {"type": "string"},
                            "designation": {"type": "string"},
                            "is_disabled": {"type": "boolean"}
                        }
                    }
                },
                "training_programs": {
                    "type": "array",
                    "items": {
                        "type": "object",
                        "properties": {
                            "name": {"type": "string"},
                            "category": {"type": "string"},
                            "cost": {"type": "number"},
                            "participants": {"type": "array"}
                        }
                    }
                },
                "suppliers": {
                    "type": "array",
                    "items": {
                        "type": "object",
                        "properties": {
                            "name": {"type": "string"},
                            "bee_level": {"type": "integer"},
                            "black_ownership": {"type": "number"},
                            "spend": {"type": "number"}
                        }
                    }
                }
            }
        }
    
    async def process_document(self, file_content: bytes, filename: str, file_type: str):
        """
        Main entry point for document processing
        """
        # Step 1: Extract text based on file type
        text = await self.extract_text(file_content, filename, file_type)
        
        # Step 2: Split text for processing
        chunks = self.text_splitter.split_text(text)
        
        # Step 3: Extract structured data using LLM
        extracted_data = await self.extract_structured_data(text)
        
        # Step 4: Validate and clean extracted data
        validated_data = self.validate_extracted_data(extracted_data)
        
        # Step 5: Generate embeddings for search
        embeddings = await self.generate_embeddings(chunks)
        
        return {
            'extracted_data': validated_data,
            'confidence': self.calculate_confidence(validated_data),
            'embeddings': embeddings,
            'warnings': self.generate_warnings(validated_data)
        }
    
    async def extract_text(self, file_content: bytes, filename: str, file_type: str) -> str:
        """Extract text from various file formats"""
        
        if file_type == 'pdf':
            # Save temp file
            with open(f'/tmp/{filename}', 'wb') as f:
                f.write(file_content)
            
            # Extract text from PDF
            loader = PyPDFLoader(f'/tmp/{filename}')
            documents = loader.load()
            text = "\n".join([doc.page_content for doc in documents])
            
            # Also do OCR for scanned PDFs
            images = pdf2image.convert_from_bytes(file_content)
            ocr_text = ""
            for image in images:
                ocr_text += pytesseract.image_to_string(image) + "\n"
            
            return text + "\n" + ocr_text
        
        elif file_type in ['jpg', 'jpeg', 'png']:
            # OCR for images
            image = Image.open(io.BytesIO(file_content))
            text = pytesseract.image_to_string(image)
            return text
        
        elif file_type == 'excel':
            # Save temp file
            with open(f'/tmp/{filename}', 'wb') as f:
                f.write(file_content)
            
            # Extract from Excel
            loader = UnstructuredExcelLoader(f'/tmp/{filename}')
            documents = loader.load()
            return "\n".join([doc.page_content for doc in documents])
        
        elif file_type in ['word', 'ppt']:
            # Use Unstructured for Word/PPT
            with open(f'/tmp/{filename}', 'wb') as f:
                f.write(file_content)
            
            from unstructured.partition.auto import partition
            elements = partition(filename=f'/tmp/{filename}')
            return "\n".join([str(el) for el in elements])
        
        else:
            raise ValueError(f"Unsupported file type: {file_type}")
    
    async def extract_structured_data(self, text: str) -> dict:
        """Use LLM to extract structured data from text"""
        
        chain = create_extraction_chain(self.llm, self.extraction_schema)
        result = chain.run(text)
        
        return result
    
    def validate_extracted_data(self, data: dict) -> dict:
        """Validate and clean extracted data"""
        validated = {}
        
        # Validate company info
        if 'company_name' in data:
            validated['company_name'] = data['company_name']
        
        if 'revenue' in data:
            try:
                validated['revenue'] = float(data['revenue'])
            except:
                validated['revenue'] = None
        
        # Validate shareholders
        if 'shareholders' in data and isinstance(data['shareholders'], list):
            validated['shareholders'] = []
            for s in data['shareholders']:
                shareholder = {
                    'name': s.get('name'),
                    'share_percentage': float(s.get('share_percentage', 0)) / 100 if s.get('share_percentage') else 0,
                    'race': s.get('race', '').lower(),
                    'gender': s.get('gender', '').lower()
                }
                validated['shareholders'].append(shareholder)
        
        # Validate employees
        if 'employees' in data and isinstance(data['employees'], list):
            validated['employees'] = []
            for e in data['employees']:
                employee = {
                    'name': e.get('name'),
                    'race': e.get('race', '').lower(),
                    'gender': e.get('gender', '').lower(),
                    'designation': e.get('designation', '').lower(),
                    'is_disabled': e.get('is_disabled', False)
                }
                validated['employees'].append(employee)
        
        return validated
    
    def calculate_confidence(self, data: dict) -> float:
        """Calculate confidence score for extraction"""
        total_fields = 0
        filled_fields = 0
        
        def count_fields(obj):
            nonlocal total_fields, filled_fields
            if isinstance(obj, dict):
                for k, v in obj.items():
                    total_fields += 1
                    if v not in [None, '', [], {}]:
                        filled_fields += 1
                    if isinstance(v, (dict, list)):
                        count_fields(v)
            elif isinstance(obj, list):
                for item in obj:
                    count_fields(item)
        
        count_fields(data)
        
        return filled_fields / max(total_fields, 1)
    
    def generate_warnings(self, data: dict) -> list:
        """Generate warnings about extracted data"""
        warnings = []
        
        # Check for missing critical fields
        if not data.get('company_name'):
            warnings.append("Company name not found")
        
        if not data.get('revenue'):
            warnings.append("Revenue not found or could not be extracted")
        
        # Check shareholder data
        shareholders = data.get('shareholders', [])
        if shareholders:
            total_ownership = sum(s.get('share_percentage', 0) for s in shareholders)
            if abs(total_ownership - 100) > 1:
                warnings.append(f"Shareholder ownership sums to {total_ownership}% (should be 100%)")
        
        return warnings
    
    async def generate_embeddings(self, chunks: list) -> list:
        """Generate embeddings for vector search"""
        embeddings = []
        for chunk in chunks:
            embedding = self.embeddings.embed_query(chunk)
            embeddings.append(embedding)
        return embeddings

    def compare_scenarios(self, scenario_id_1: str, scenario_id_2: str) -> dict:
        """
        Compare two scenarios side by side
        """
        from calculators.summary import SummaryCalculator
        
        calculator = SummaryCalculator(self.db)
        
        # Get both scorecards
        scorecard1 = calculator.calculate_final_scorecard(None, scenario_id_1)
        scorecard2 = calculator.calculate_final_scorecard(None, scenario_id_2)
        
        # Calculate differences
        differences = {
            'total_points': scorecard2['total_points'] - scorecard1['total_points'],
            'level': scorecard1['achieved_level'] - scorecard2['achieved_level'],
            'pillars': {}
        }
        
        for pillar in scorecard1['pillar_breakdown'].keys():
            diff = scorecard2['pillar_breakdown'][pillar] - scorecard1['pillar_breakdown'][pillar]
            differences['pillars'][pillar] = diff
        
        # Calculate impact rating
        if differences['level'] > 0:
            impact = "Positive"
        elif differences['level'] < 0:
            impact = "Negative"
        else:
            impact = "Neutral"
        
        return {
            'scenario_1': {
                'id': scenario_id_1,
                'name': scorecard1['scenario_name'],
                'scorecard': scorecard1
            },
            'scenario_2': {
                'id': scenario_id_2,
                'name': scorecard2['scenario_name'],
                'scorecard': scorecard2
            },
            'differences': differences,
            'impact': impact,
            'investment_required': self._calculate_investment_difference(scenario_id_1, scenario_id_2)
        }
    
    def _calculate_investment_difference(self, scenario_id_1: str, scenario_id_2: str) -> dict:
        """Calculate investment difference between scenarios"""
        
        # Get training spend difference
        training1 = self.db.query(TrainingPrograms).filter(
            TrainingPrograms.scenario_id == scenario_id_1
        ).all()
        training2 = self.db.query(TrainingPrograms).filter(
            TrainingPrograms.scenario_id == scenario_id_2
        ).all()
        
        training_spend1 = sum(t.cost for t in training1)
        training_spend2 = sum(t.cost for t in training2)
        
        # Get procurement spend difference
        proc1 = self.db.query(ProcurementTransactions).filter(
            ProcurementTransactions.scenario_id == scenario_id_1
        ).all()
        proc2 = self.db.query(ProcurementTransactions).filter(
            ProcurementTransactions.scenario_id == scenario_id_2
        ).all()
        
        proc_spend1 = sum(t.amount for t in proc1)
        proc_spend2 = sum(t.amount for t in proc2)
        
        # Get ESD/SED spend difference
        esd1 = self.db.query(ESDContributions).filter(
            ESDContributions.scenario_id == scenario_id_1
        ).all()
        esd2 = self.db.query(ESDContributions).filter(
            ESDContributions.scenario_id == scenario_id_2
        ).all()
        
        esd_spend1 = sum(c.amount for c in esd1)
        esd_spend2 = sum(c.amount for c in esd2)
        
        return {
            'training': training_spend2 - training_spend1,
            'procurement': proc_spend2 - proc_spend1,
            'esd_sed': esd_spend2 - esd_spend1,
            'total': (training_spend2 + proc_spend2 + esd_spend2) - 
                     (training_spend1 + proc_spend1 + esd_spend1)
        }

    async def export_verification_zip(self, client_id: str, scenario_id: str) -> bytes:
        """Generate verification pack ZIP with all supporting documents"""
        
        zip_buffer = io.BytesIO()
        
        with zipfile.ZipFile(zip_buffer, 'w', zipfile.ZIP_DEFLATED) as zip_file:
            
            # Add scorecard PDF
            scorecard_pdf = await self.export_scorecard_pdf(client_id, scenario_id)
            zip_file.writestr('scorecard.pdf', scorecard_pdf)
            
            # Add gap analysis PDF
            gap_pdf = await self.export_gap_analysis_pdf(client_id, scenario_id)
            zip_file.writestr('gap_analysis.pdf', gap_pdf)
            
            # Add audit Excel
            audit_excel = await self.export_audit_excel(client_id, scenario_id)
            zip_file.writestr('audit_data.xlsx', audit_excel)
            
            # Add all uploaded documents from Data Hub
            documents = self.db.query(Document).filter(
                Document.client_id == client_id
            ).all()
            
            for doc in documents:
                # Download from S3 and add to zip
                import boto3
                s3 = boto3.client('s3')
                obj = s3.get_object(Bucket='okiru-documents', Key=doc.s3_key)
                zip_file.writestr(f'documents/{doc.filename}', obj['Body'].read())
            
            # Add verification summary JSON
            from calculators.summary import SummaryCalculator
            calculator = SummaryCalculator(self.db)
            scorecard = calculator.calculate_final_scorecard(client_id, scenario_id)
            
            verification_summary = {
                'client_id': str(client_id),
                'scenario_id': str(scenario_id),
                'generated_at': datetime.now().isoformat(),
                'scorecard': scorecard,
                'verification_ready': True,
                'documents_included': len(documents)
            }
            
            zip_file.writestr('verification_summary.json', 
                             json.dumps(verification_summary, indent=2, default=str))
        
        zip_buffer.seek(0)
        return zip_buffer.getvalue()
    
    def _get_pillar_target(self, pillar: str) -> float:
        """Get target points for a pillar"""
        targets = {
            'ownership': 25,
            'management': 19,
            'skills': 25,
            'procurement': 29,
            'esd': 15,
            'sed': 5,
            'yes': 3
        }
        return targets.get(pillar, 0)

  points: number;
  target: number;

export function PillarCard({ name, points, target, onClick }: PillarCardProps) {

  const getColor = (percent: number) => {

    if (percent >= 40) return 'bg-yellow-500';

      onClick={onClick}

          <span className="text-2xl font-bold">{points.toFixed(1)}</span>
          <span className="text-gray-500">/ {target}</span>
        </div>
        <Progress 
          value={percentage} 
          className={`h-2 ${getColor(percentage)}`}
        />
        <div className="text-right mt-2 text-sm text-gray-600">
          {percentage.toFixed(1)}%
        </div>
      </CardContent>
    </Card>
  );

      onUpload(acceptedFiles[0]);
    }
  }, [onUpload]);

      'application/pdf': ['.pdf'],
      'image/*': ['.jpg', '.jpeg', '.png']
    },
    maxFiles: 1
  });

  return (
    <div
      {...getRootProps()}
      className={`border-2 border-dashed rounded-lg p-12 text-center cursor-pointer transition-colors
        ${isDragActive ? 'border-blue-500 bg-blue-50' : 'border-gray-300 hover:border-gray-400'}`}
    >
      <input {...getInputProps()} />
      <Upload className="mx-auto h-12 w-12 text-gray-400 mb-4" />

        <p className="text-lg">Drop the file here...</p>

function ComparisonView({ scenarioId, baseId }: { scenarioId: string, baseId: string }) {
  const { data: comparison } = useQuery({
    queryKey: ['comparison', scenarioId, baseId],
    queryFn: async () => {
      const res = await fetch(`/api/scenarios/compare?id1=${baseId}&id2=${scenarioId}`);
      return res.json();
    }
  });

  if (!comparison) return null;

  return (
    <Card>
      <CardHeader>
        <CardTitle>Scenario Comparison</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="grid grid-cols-3 gap-4 text-center mb-4">
          <div>
            <div className="text-sm text-gray-500">Base</div>
            <div className="text-2xl font-bold">Level {comparison.scenario_1.scorecard.achieved_level}</div>
            <div className="text-sm">{comparison.scenario_1.scorecard.total_points.toFixed(2)} pts</div>
          </div>
          <div>
            <div className="text-sm text-gray-500">Difference</div>
            <div className={`text-2xl font-bold ${
              comparison.differences.level > 0 ? 'text-green-600' : 
              comparison.differences.level < 0 ? 'text-red-600' : 'text-gray-600'
            }`}>
              {comparison.differences.level > 0 ? '+' : ''}{comparison.differences.level} level
            </div>
            <div className="text-sm">
              {comparison.differences.total_points > 0 ? '+' : ''}
              {comparison.differences.total_points.toFixed(2)} pts
            </div>
          </div>
          <div>
            <div className="text-sm text-gray-500">Scenario</div>
            <div className="text-2xl font-bold">Level {comparison.scenario_2.scorecard.achieved_level}</div>
            <div className="text-sm">{comparison.scenario_2.scorecard.total_points.toFixed(2)} pts</div>
          </div>
        </div>

        <div className="space-y-2">
          <h4 className="font-semibold">Pillar Differences:</h4>
          {Object.entries(comparison.differences.pillars).map(([pillar, diff]) => (
            <div key={pillar} className="flex justify-between items-center">
              <span className="capitalize">{pillar}</span>
              <span className={Number(diff) > 0 ? 'text-green-600' : Number(diff) < 0 ? 'text-red-600' : 'text-gray-600'}>
                {Number(diff) > 0 ? '+' : ''}{Number(diff).toFixed(2)}
              </span>
            </div>
          ))}
        </div>

        {comparison.investment_required.total !== 0 && (
          <div className="mt-4 p-3 bg-gray-50 rounded">
            <div className="font-semibold">Investment Required:</div>
            <div className="text-lg font-bold">
              R {comparison.investment_required.total.toLocaleString()}
            </div>
          </div>
        )}
      </CardContent>
    </Card>
  );
}
